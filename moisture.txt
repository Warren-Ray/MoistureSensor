int sensorPin = A0; 
int sensorValue;  
int limit = 300; 

void setup() {
 Serial.begin(9600);
 pinMode(13, OUTPUT);
  pinMode(10, OUTPUT);
}

void loop() {

 sensorValue = analogRead(sensorPin); 
 Serial.println("Analog Value : ");
 Serial.println(sensorValue);
 
 if (sensorValue<limit) {
 digitalWrite(13, HIGH); 
   digitalWrite(10, LOW);
 }
 else {
 digitalWrite(13, LOW); 
   digitalWrite(10, HIGH);
 }
 
 delay(1000); 
}
